# nano ~/infrastructure/docker-compose.yml
services:
  portfolio:
    image: nginx:alpine
    container_name: portfolio
    restart: unless-stopped
    volumes:
      - ~/sites/profile_website:/usr/share/nginx/html:ro
      - ~/infrastructure/nginx/portfolio.conf:/etc/nginx/conf.d/default.conf:ro
      - ~/infrastructure/nginx/certs:/etc/nginx/certs:ro
    networks:
      - webnet

  project:
    image: nginx:alpine
    container_name: project
    restart: unless-stopped
    volumes:
      - ~/sites/cyber_law_project/dist:/usr/share/nginx/html:ro
      - ~/infrastructure/nginx/project.conf:/etc/nginx/conf.d/default.conf:ro
      - ~/infrastructure/nginx/certs:/etc/nginx/certs:ro
    networks:
      - webnet

  cloudflared:
    image: cloudflare/cloudflared:latest
    container_name: cloudflared
    restart: unless-stopped
    command: tunnel run
    environment:
      - TUNNEL_TOKEN=${CLOUDFLARE_TUNNEL_TOKEN}
    networks:
      - webnet

networks:
  webnet:
    driver: bridge
